local function touchDeck( event )
	if event.phase == "began" then
		if gameState == "gameover" then
			gameState = "swap"

			instructions.text = "Choose which cards you wish to swap."
			deck.canTouchCard = true

			-- Sekoitetaan pakka.
			deck.shuffle()

			-- Nostetaan 5 uutta korttia.
			for i = 1, 5 do
				-- Poistetaan vanhat kortit aina ennen
				-- kuin pelaajalle jaetaan uudet.
				display.remove( hand[i] )

				hand[i] = deck.draw( i, gameDeck )
			end

		elseif gameState == "swap" then
			gameState = "gameover"

			instructions.text = "Tap the deck to draw new cards."
			deck.canTouchCard = false

			-- Käydään pelaajan kädessä olevat 5
			-- korttia läpi ja tuhotaan valituttuna
			-- olevat kortit ja nostetaan niiden
			-- tilalle uudet kortit.
			for i = 1, 5 do
				-- deck.lua:ssa asetamme jokaiselle kortille
				-- isSelected = true, jos ne ovat valittuna.
				if hand[i].isSelected then
					display.remove( hand[i] )

					hand[i] = deck.draw( i, gameDeck )
				end
			end

			-- Tarkistetaan pelaajan käsi.
			checkHand.get( hand )

		end
	end

	return true
end
